160 void Neuron::updateActivity()
161 {
162  double newActivity = bias + input;
163  for (SynapseList::iterator it = synapsesIn.begin(); it != synapsesIn.end(); it++)
164  {
165  newActivity += it->second->getWeight() * it->first->getOutput();
166  }

170 void Neuron::updateError()
171 {

174  it++)

177  }
