409 bool ANN::updateTopologicalSort()
410 {
411  typedef std::set<Neuron*> NeuronSet;
412  typedef std::vector<Synapse*> SynapseVector;
413  typedef std::map<Neuron*, NeuronSet> SynapticMap;
414 
415  topologicalSort.clear();
416  NeuronList neurons = getAllNeurons();
417  SynapticMap synapsesIn;
418  SynapticMap synapsesOut;
419  NeuronSet noIncomes;
420 
421  // run over all neurons and pouplate the containers
422  for (NeuronList::iterator neuronIt = neurons.begin(); neuronIt!=neurons.end();
423  neuronIt++)
424  {
425  Neuron* const n = *neuronIt;
426  SynapseVector ins = n->getSynapsesIn();
427  SynapseVector outs = n->getSynapsesOut();
428  for (SynapseVector::iterator sIt = ins.begin(); sIt!=ins.end(); sIt++)
429  synapsesIn[n].insert((*sIt)->getPre());
430  for (SynapseVector::iterator sIt = outs.begin(); sIt!=outs.end(); sIt++)
431  synapsesOut[n].insert((*sIt)->getPost());
432  if (ins.size()==0) noIncomes.insert(n);
433  }
434 
435  // start the algorithm
436  while (noIncomes.size()>0)
437  {
438  Neuron* n = (*noIncomes.begin());
439  noIncomes.erase(n);
440  topologicalSort.push_back(n);
441  while (synapsesOut.find(n) != synapsesOut.end())
442  {
443  Neuron* post = *(synapsesOut[n].begin());
444  synapsesOut[n].erase(post);
445  synapsesIn[post].erase(n);
446  if (synapsesIn[post].size()==0) noIncomes.insert(post);
447  if (synapsesOut[n].size()==0) synapsesOut.erase(n);
448  if (synapsesIn[post].size()==0) synapsesIn[post].erase(post);
449  }
450  }
451 
452  return (synapsesOut.size()==0);
453 }
