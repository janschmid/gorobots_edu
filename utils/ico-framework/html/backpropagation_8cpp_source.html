73 void Backpropagation::learn(const unsigned int& epochs)
74 {
75  typedef std::map<Synapse*, double> WeightMap;
76  typedef std::map<Neuron*, double> BiasMap;
77 
78  // open file to log error function
79  std::ofstream errorFunctionFile("errorFunction.dat");
80 
81  // iterate over epochs
82  for (unsigned int e=0; e<epochs; e++)
83  {
84  WeightMap weightChange;
85  BiasMap biasChange;
86 
87  // variable to calculate the error function in this epoch
88  double epochErrorFunction = 0;
89 
90  // iterate over all patterns
91  for (PatternVector::iterator patternIt = patterns.begin();
92  patternIt < patterns.end(); patternIt++)
93  {
94 
95  // set the neuron inputs
96  for (NeuronMap::iterator it = inputNeurons.begin(); it!=inputNeurons.end();
97  it++)
98  {
99  it->second->setInput((*patternIt)->inputs[it->first]);
100  }
101 
102  // perform the feed forward step
103  net->feedForwardStep();
104 
105  // calculate error signal for output neurons
106  double patternError = 0;
107  for (NeuronMap::iterator it = outputNeurons.begin();
108  it!=outputNeurons.end(); it++)
109  {
110  const double output = net->getOutput(it->second);
111  const double target = (*patternIt)->outputs[it->first];
112  const double error = output-target;
113  it->second->setErrorInput(error);
114  patternError += error * error;
115  }
116  epochErrorFunction += 0.5*patternError;
117 
118  // perform backpropagation step
119  net->backpropagationStep();
120 
121  // calculate synaptic weight changes for all synapses
122  for (SynapseSet::iterator it = synapses.begin(); it!=synapses.end(); it++)
123  {
124  const double dw = -(*it)->getPre()->getOutput() * (*it)->getPost()->getError();
125  weightChange[*it] += dw;
126  }
127 
128  // update neuron biases
129  for (NeuronSet::iterator it = neurons.begin(); it!=neurons.end(); it++)
130  {
131  const double db = -(*it)->getError();
132  biasChange[*it] += db;
133  }
134 
135  }
136 
137  // apply weight changes
138  for (SynapseSet::iterator it = synapses.begin(); it!=synapses.end(); it++)
139  {
140  (*it)->setWeight( (*it)->getWeight() + rate * weightChange[*it]);
141  }
142 
143  // apply bias changes
144  for (NeuronSet::iterator it = neurons.begin(); it!=neurons.end(); it++)
145  {
146  (*it)->setBias( (*it)->getBias() + rate * biasChange[*it]);
147  }
148 
149  errorFunctionFile << e << "\t" << epochErrorFunction << std::endl;
150  }
151 }
